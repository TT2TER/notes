## const和数组
```c
#include<stdio.h>
#include<string.h>
int main()
{
    int n;
    scanf("%d",&n);
    const int N=/*(const int)*/n;//注释有无均可
    char arr[N];
    int i;
    for(i=0;i<N;i++)
    {
	arr[i]='c';
    printf("%c",arr[i]);
	}
    return 0;
}
```
